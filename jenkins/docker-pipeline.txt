pipeline {

  environment {
    registry = "mrmartyn05/profile-service"
    registryCredential = 'dockerhub'
    dockerImage = ''

  }
  
  agent any
  
  stages {
    
    stage('Cloning Git') {
      steps {
        git 'https://github.com/muttbag/profile-service.git'
      }
    }
    
    stage('Copy artifact'){
      steps {
          copyArtifacts fingerprintArtifacts: true, projectName: 'profile-service-build/master', selector: lastSuccessful(), target: '$WORKSPACE'
      }        
    }
    
    stage('Building image') {
      steps{
        script {
          dockerImage = docker.build '-f directory/file' registry + ":$BUILD_NUMBER"
        }
      }
    }
    
    stage('Deploy Image') {
      steps{
        script {
          docker.withRegistry( '', registryCredential ) {
            dockerImage.push()
          }
        }
      }
    }
    
    stage('Remove Unused docker image') {
        steps{
        sh "docker rmi $registry:$BUILD_NUMBER"
    }
}
    stage('Deploy Stack') {
      steps {
            script {
                echo "Deploying Container Stack to Docker Cluster"
                sh "ansible-playbook $WORKSPACE/ansible/update_docker_swarm.yml --extra-vars \"PROJECT=telecom DOCKER_USER=mrmartyn05 DOCKER_PASSWORD=F4i&t7ME6Es!Hr8b$cjKto!wQV*1LI\" -vvv"
            }
          }
      }
    }

  }
